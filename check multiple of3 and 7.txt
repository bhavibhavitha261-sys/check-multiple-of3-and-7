#include <stdio.h>

int main() {
    int num;

    printf("Enter a number: ");
    scanf("%d", &num);

    if (num % 3 == 0 && num % 7 == 0) {
        printf("The number is a multiple of both 3 and 7\n");
    } else {
        printf("The number is NOT a multiple of both 3 and 7\n");
    }

    return 0;
}
